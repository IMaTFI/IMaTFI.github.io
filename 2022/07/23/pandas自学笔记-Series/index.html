创建

# 通过列表创建Series
a = [1, 2, 3, 4, 5]
t = pd.Series(a)
print(t)
# 0    1
# 1    2
# 2    3
# 3    4
# 4    5
# dtype: int64

# 指定索引
a = [1, 2, 3, 4, 5]
t = pd.Series(a, index=list("abcde"))
print(t)
# a    1
# b    2
# c    3
# d    4
# e    5
# dtype: int64

# 通过字典创建Series
a = {"姓名": "张三", "年龄": "17", "性别": "男"}
t = pd.Series(a, index=["姓名", "年龄", "身高"])
print(t)
# 姓名     张三
# 年龄     17
# 身高    NaN		# 索引匹配失败的元素为NaN
# dtype: object
